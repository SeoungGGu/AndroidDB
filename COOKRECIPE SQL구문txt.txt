CREATE TABLE m_user (
  USER_ID VARCHAR(255) NOT NULL,
  PW VARCHAR(255) NOT NULL,
  Name VARCHAR(255) NOT NULL,
  Mail VARCHAR(255) NOT NULL,
  PRIMARY KEY (USER_ID)
);

CREATE TABLE INGREDIENT (
  INGREDIENT_ID INT(255) NOT NULL AUTO_INCREMENT,
  USER_ID VARCHAR(255),
  Name VARCHAR(255) NOT NULL,
  Category VARCHAR(255),
  PRIMARY KEY (INGREDIENT_ID),
  ImagePath VARCHAR(255),
  FOREIGN KEY (USER_ID) REFERENCES m_user(USER_ID)
);

CREATE TABLE RECIPE (
  USER_ID VARCHAR(255),
  INGREDIENT_ID INT(255),
  RECIPE_ID INT NOT NULL AUTO_INCREMENT,
  Title VARCHAR(255) NOT NULL,
  Category VARCHAR(255) NOT NULL,
  View INTEGER,
  Difficulty INTEGER,
  Servings INTEGER,
  Contents TEXT NOT NULL,
  ImagePath VARCHAR(255),
  PRIMARY KEY (RECIPE_ID),
  FOREIGN KEY (USER_ID) REFERENCES m_user(USER_ID),
  FOREIGN KEY (INGREDIENT_ID) REFERENCES INGREDIENT(INGREDIENT_ID)
);

CREATE TABLE NOTICE (
   noticeContent VARCHAR(1000) NOT NULL,
   noticecName VARCHAR(50) NOT NULL,
   noticeDate DATE NOT NULL
);

CREATE TABLE ScrapRecipe (
    USER_ID VARCHAR(255),
    RECIPE_ID INT NOT NULL,
    ScrapRecipe_ID INT NOT NULL AUTO_INCREMENT,
    PRIMARY KEY (ScrapRecipe_ID),
    FOREIGN KEY (USER_ID) REFERENCES m_user(USER_ID),
    FOREIGN KEY (RECIPE_ID) REFERENCES RECIPE(RECIPE_ID)
);
